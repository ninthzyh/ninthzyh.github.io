1.[[html]为什么HTML5只需要写<!DOCTYPE HTML>就可以？]

html5不基于标准通用标记语言（SGML），不用对DTD引用，需要DOCTYPE规范浏览器行为（按照什么类型运行）；

html4基于SGML，需要对DTD引用，才能知道文档类型（document type）HTML版本

2.[[css]CSS选择器有哪些？哪些属性可以继承？]

选择器
标签、id、class
结构 1）后代（空格）
2）子元素（>）
3）兄弟（~）
4）铁兄弟（+）
属性 1）[key]
2）[key=value]
3）[key^=value]以value开头
4）[key$=value]以value结尾
5）[key*=value]有value
6）[key~=value]value是一个单词
7）[key|=value]开头value，‘-’连接
伪类 1）:link初始值
2）:visited点击后
3）:hover悬浮
4）:acitve点击中
5）:focus聚焦
6）:target目标
7）:empty空
8）:root根
伪类元素 1）:first-child|:last-child选择第一个|最后一个元素
2）:first-of-type|:last-of-type选择第一个|最后一个元素类型
3）:only-child唯一元素
4）:only-of-type唯一元素类型
5）:nth-child(n)第n个元素
6）:nth-last-child(n)最后第n个元素
7）:nth-last-of-type(n)最后第n个元素类型
8）:not(n)排除第n个元素
表单伪类 1）:disabled禁用
2）:enabled可用
3）:required必填
4）:optional非必填
5）:checked选中
文本伪类 1）::first-letter选择文本第一个字
2）::first-line选择文本第一行
3）::after|::before选择元素后面|前面的元素
可继承 不可继承
font|color margin|padding|border|width|height

3.[[js]写一个方法把下划线命名转成大驼峰命名]

function bigHumpFunc(str) {
    // str前没有‘_’,给str加上
    if (str[0] !== "_") str = "_" + str;
    let arr = str.split("");
    arr.map((item, index) => {
        // 有‘_’，删除‘_’，删除后当前索引item转换大写
        if (item === "_") {
            arr.splice(index, 1);
            arr[index] = arr[index].toUpperCase();
        }
    });
    return arr.join("");
}
let hump = bigHumpFunc("_ad_bd_cd");
console.log(hump);
function bigHumpFunc(str) {
    let arr = str.split('_').map((item,index)=>(item[0]) && item[0].toUpperCase()+item.substr(1))
    return arr.join('')
}
